    if (numero < 1 || numero > 99999) {
        System.out.println("El número debe estar entre 1 y 99999");
        return;
    }
    String numeroComoCadena = Integer.toString(numero);
    for (int i = 0; i < numeroComoCadena.length(); i++) {
        System.out.print(numeroComoCadena.charAt(i));
        if (i < numeroComoCadena.length() - 1) {
            System.out.print("  ");
        }
    }
    System.out.println();
}
